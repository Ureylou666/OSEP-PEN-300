	Dim str As string

	Dim exePath As String
	exePath = ActiveDocument.Path + "\evil.exe"

End Sub
-----------------------------------------
Sub Wait(n as Long)
	Dim t as Date
	t = Now
	Do 
		DoEvents
	Loop Until Now >= DateAdd("s",n,t)
End Sub
